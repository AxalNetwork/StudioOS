## 📁 `server.ts`

```ts
import express from "express";

const app = express();
app.use(express.json());

type StartupInput = {
  market: {
    tam: number;
    urgency: number; // 0–10
    trend: number;   // 0–5
  };
  team: {
    experience: number; // 0–8
    execution: number;  // 0–8
    network: number;    // 0–4
  };
  product: {
    mvpTimeDays: number;
    complexity: number; // 0–5 (lower better)
    dependencies: number; // 0–3
  };
  financials: {
    costToMVP: number;
    timeToRevenueMonths: number;
    burnRisk: number; // 0–3
  };
  axalFit: number; // 0–15
  distribution: {
    channels: number; // 0–5
    virality: number; // 0–5
  };
};

function scoreMarket(m: StartupInput["market"]) {
  let score = 0;

  if (m.tam > 1_000_000_000) score += 10;
  else if (m.tam > 100_000_000) score += 7;
  else score += 3;

  score += m.urgency;
  score += m.trend;

  return Math.min(score, 25);
}

function scoreTeam(t: StartupInput["team"]) {
  return Math.min(t.experience + t.execution + t.network, 20);
}

function scoreProduct(p: StartupInput["product"]) {
  let score = 0;

  if (p.mvpTimeDays <= 30) score += 7;
  else if (p.mvpTimeDays <= 60) score += 5;
  else score += 2;

  score += (5 - p.complexity); // lower complexity = better
  score += (3 - p.dependencies);

  return Math.min(score, 15);
}

function scoreCapital(f: StartupInput["financials"]) {
  let score = 0;

  if (f.costToMVP < 50000) score += 7;
  else if (f.costToMVP < 200000) score += 5;
  else score += 2;

  if (f.timeToRevenueMonths <= 6) score += 5;
  else if (f.timeToRevenueMonths <= 12) score += 3;
  else score += 1;

  score += (3 - f.burnRisk);

  return Math.min(score, 15);
}

function scoreDistribution(d: StartupInput["distribution"]) {
  return Math.min(d.channels + d.virality, 10);
}

function classify(score: number) {
  if (score >= 85) return "TIER_1";
  if (score >= 70) return "TIER_2";
  return "REJECT";
}

app.post("/scoreStartup", (req, res) => {
  const data: StartupInput = req.body;

  const market = scoreMarket(data.market);
  const team = scoreTeam(data.team);
  const product = scoreProduct(data.product);
  const capital = scoreCapital(data.financials);
  const fit = data.axalFit;
  const distribution = scoreDistribution(data.distribution);

  const total =
    market * 0.25 +
    team * 0.2 +
    product * 0.15 +
    capital * 0.15 +
    fit * 0.15 +
    distribution * 0.1;

  const tier = classify(total);

  res.json({
    score: Math.round(total),
    tier,
    breakdown: {
      market,
      team,
      product,
      capital,
      fit,
      distribution,
    },
  });
});

app.listen(3000, () => {
  console.log("Axal Scoring API running on port 3000");
});
```
