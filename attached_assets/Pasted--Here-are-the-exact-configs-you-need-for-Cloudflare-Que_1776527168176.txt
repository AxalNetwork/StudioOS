### 2. New consumer Worker (create this file)

Create `cloudflare-worker/queue-consumer.js`:

```js
export default {
  async queue(batch, env) {
    for (const message of batch.messages) {
      const { job_type, payload } = message.body;

      try {
        console.log(`Processing job: ${job_type}`, payload);

        if (job_type === 'ai_scoring') {
          // Call your scoring-worker
          await env.AI.run('@cf/meta/llama-3-8b-instruct', { /* prompt */ });
        } else if (job_type === 'spinout_execute') {
          // Call spin-out logic
        } else if (job_type === 'traction_review') {
          // etc.
        }

        message.ack(); // Mark as processed
      } catch (err) {
        console.error(`Job failed: ${job_type}`, err);
        // Optional: retry or dead-letter
      }
    }
  }
};
```

### 3. How to send jobs from FastAPI (example)

In `backend/jobs.py` (or wherever you enqueue):

```python
await env.JOB_QUEUE.send({
    "job_type": "spinout_execute",
    "payload": { "deal_id": 123, "user_id": 456 }
})
```
